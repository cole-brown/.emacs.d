(use-package flyspell
  :init
  (setq ispell-dictionary "en_US")
  (setq ispell-program-name (executable-find "hunspell"))
  (setq ispell-really-hunspell t)
  :hook (prog-mode . flyspell-prog-mode)
  (text-mode . flyspell-mode))
